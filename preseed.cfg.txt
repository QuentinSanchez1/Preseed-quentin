# Démarrage automatique de l'installation
d-i debian-installer/locale string fr_FR.UTF-8
d-i console-setup/ask_detect boolean false
d-i keyboard-configuration/xkb-keymap select fr

# Configuration réseau
d-i netcfg/choose_interface select auto
d-i netcfg/get_hostname string ubuntu
d-i netcfg/get_domain string localdomain

# Configuration des utilisateurs
d-i passwd/root-login boolean false
d-i passwd/make-user boolean true
d-i passwd/user-fullname string Quentin
d-i passwd/username string quentin
d-i passwd/user-password password quentin
d-i passwd/user-password-again password quentin

# Configuration du disque
d-i partman-auto/method string lvm
d-i partman-lvm/device_remove_lvm boolean true
d-i partman-lvm/confirm boolean true
d-i partman-auto/choose_recipe select atomic
d-i partman/confirm boolean true
d-i partman/confirm_nooverwrite boolean true

# Installation des paquets
tasksel tasksel/first multiselect standard, ubuntu-desktop
d-i pkgsel/include string openssh-server

# Mise à jour des paquets
d-i pkgsel/upgrade select safe-upgrade

# Fin de l'installation
d-i grub-installer/only_debian boolean true
d-i finish-install/reboot_in_progress note

# Configuration du fuseau horaire
d-i time/zone string Europe/Paris
